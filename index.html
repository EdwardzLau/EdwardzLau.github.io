import numpy as np
import databricks.sql as dbsql
query = """
SELECT user_id, device_id, amount, created_at
FROM fintech_ledger.transactions
WHERE status = 'APPROVED' AND created_at >= '2025-01-01'
"""
with dbsql.connect(properties) as conn:
df = pd.read_sql(query, conn)
